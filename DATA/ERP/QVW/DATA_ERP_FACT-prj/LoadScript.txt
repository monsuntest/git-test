///$tab Main
// SE INLUYEN LAS VARIABLES DE FORMATO

$(Include=..\cfg\variables.txt);

X_ReloadTime:
LOAD
	NOW() 		AS StartTimes,
	OSUSER()	AS ReloadUser
AUTOGENERATE (1);

// SE INCLUYEN LAS VARIABLES DE CONFIGURACION
$(Include=..\cfg\config.txt);

// NOMBRAR AQUI LA TABLA
SET S_TABLA = 'FACT';

SET S_TABLA_2 = 'NACHITO';




///$tab TABLA
// CARGAR LA TABLA
$(S_TABLA):
LOAD ID_TRANSACCION, 
     ID_PRODUCTO, 
     ID_CLIENTE, 
     TIPO_VTA, 
     FECHA, 
     Importe
FROM
[..\..\..\FUENTES_MOCK\FACT.xls]
(biff, embedded labels, table is FACT$);











///$tab STORE
// GUARDAR AQUI LA TABLA

STORE $(S_TABLA) INTO ..\QVD\DATA_ERP_$(S_TABLA).QVD;

DROP TABLE $(S_TABLA);

///$tab END
JOIN (X_ReloadTime)
LOAD 
	NOW() 								AS EndTime 
AUTOGENERATE(1);

tExecTime:
LOAD
	TIME(EndTime-StartTime,'hh:mm:ss') 	AS execTime
RESIDENT X_ReloadTime;
